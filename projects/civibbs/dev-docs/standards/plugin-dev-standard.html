2. plugin.py — 插件主逻辑

PYTHON
"""
FS-01 create_directory 插件
版本: 1.0.0
描述: 创建目录（如不存在）

依赖: 标准库 pathlib, logging（无第三方依赖）
"""

from __future__ import annotations

import logging
from dataclasses import dataclass
from pathlib import Path
from typing import Optional

# ============================================================
# 常量 & 错误码
# ============================================================

PLUGIN_NAME = "create_directory"
PLUGIN_VERSION = "1.0.0"

# 路径约束
_MIN_PATH_LENGTH = 1
_MAX_PATH_LENGTH = 1024

# 错误码枚举
class ErrorCode:
    E_PERMISSION = "E_PERMISSION"
    E_PATH_INVALID = "E_PATH_INVALID"
    E_IO_ERROR = "E_IO_ERROR"


# ============================================================
# 数据模型
# ============================================================

@dataclass(frozen=True)
class CreateDirectoryResult:
    """create_directory 的返回结果。"""
    created: bool       # 本次是否创建了目录
    exists: bool        # 目录是否存在（创建后）
    error_code: Optional[str] = None   # 错误码（成功时为 None）
    error_message: Optional[str] = None  # 错误信息（成功时为 None）

    def to_dict(self) -> dict:
        """转换为字典，便于序列化。"""
        result = {
            "created": self.created,
            "exists": self.exists,
        }
        if self.error_code is not None:
            result["error_code"] = self.error_code
            result["error_message"] = self.error_message
        return result


# ============================================================
# 核心函数
# ============================================================

logger = logging.getLogger(f"libsmall.{PLUGIN_NAME}")


def create_directory(dir_path: str) -> CreateDirectoryResult:
    """
    创建目录（如不存在）。

    参数:
        dir_path (str): 要创建的目录路径。

    返回:
        CreateDirectoryResult:
            - created: 本次是否创建了目录
            - exists:  目录是否存在（创建后）
            - error_code: 错误码（成功时为 None）
            - error_message: 错误信息（成功时为 None）

    内部逻辑:
        1. 校验 dir_path（非空、长度 1-1024）
        2. 将 dir_path 转换为 Path 对象
        3. 检查路径是否已存在
           - 已存在 → created=False, exists=True
        4. 如果不存在
           - 调用 path.mkdir(parents=True, exist_ok=True)
           - created=True, exists=True
        5. 异常处理
           - PermissionError → 记录日志，返回 (False, False)
           - OSError         → 记录日志，返回 (False, False)
    """

    # ----------------------------------------------------------
    # Step 0: 输入校验
    # ----------------------------------------------------------
    if dir_path is None or not isinstance(dir_path, str):
        error_msg = f"创建目录失败: 路径不能为 None 或非字符串类型"
        logger.error(error_msg)
        return CreateDirectoryResult(
            created=False,
            exists=False,
            error_code=ErrorCode.E_PATH_INVALID,
            error_message=error_msg,
        )

    # 去除首尾空格后检查
    stripped_path = dir_path.strip()
    if len(stripped_path) < _MIN_PATH_LENGTH:
        error_msg = f"创建目录失败: 路径不能为空"
        logger.error(error_msg)
        return CreateDirectoryResult(
            created=False,
            exists=False,
            error_code=ErrorCode.E_PATH_INVALID,
            error_message=error_msg,
        )

    if len(stripped_path) > _MAX_PATH_LENGTH:
        error_msg = (
            f"创建目录失败: 路径长度 {len(stripped_path)} 超过最大限制 "
            f"{_MAX_PATH_LENGTH} 字符"
        )
        logger.error(error_msg)
        return CreateDirectoryResult(
            created=False,
            exists=False,
            error_code=ErrorCode.E_PATH_INVALID,
            error_message=error_msg,
        )

    # ----------------------------------------------------------
    # Step 1: 转换为 Path 对象
    # ----------------------------------------------------------
    try:
        path = Path(stripped_path)
    except Exception as exc:
        error_msg = f"创建目录失败 {stripped_path}, {exc}"
        logger.error(error_msg)
        return CreateDirectoryResult(
            created=False,
            exists=False,
            error_code=ErrorCode.E_PATH_INVALID,
            error_message=error_msg,
        )

    # ----------------------------------------------------------
    # Step 2: 检查路径是否已存在
    # ----------------------------------------------------------
    try:
        if path.exists() and path.is_dir():
            logger.debug(f"目录已存在 {stripped_path}")
            return CreateDirectoryResult(created=False, exists=True)
    except PermissionError:
        error_msg = f"创建目录失败 {stripped_path}, 权限不足"
        logger.error(error_msg)
        return CreateDirectoryResult(
            created=False,
            exists=False,
            error_code=ErrorCode.E_PERMISSION,
            error_message=error_msg,
        )
    except OSError as exc:
        error_msg = f"创建目录失败 {stripped_path}, {exc}"
        logger.error(error_msg)
        return CreateDirectoryResult(
            created=False,
            exists=False,
            error_code=ErrorCode.E_IO_ERROR,
            error_message=error_msg,
        )

    # ----------------------------------------------------------
    # Step 3: 目录不存在，执行创建
    # ----------------------------------------------------------
    try:
        path.mkdir(parents=True, exist_ok=True)
        logger.debug(f"目录创建成功 {stripped_path}")
        return CreateDirectoryResult(created=True, exists=True)

    except PermissionError:
        error_msg = f"创建目录失败 {stripped_path}, 权限不足"
        logger.error(error_msg)
        return CreateDirectoryResult(
            created=False,
            exists=False,
            error_code=ErrorCode.E_PERMISSION,
            error_message=error_msg,
        )

    except OSError as exc:
        error_msg = f"创建目录失败 {stripped_path}, {exc}"
        logger.error(error_msg)
        return CreateDirectoryResult(
            created=False,
            exists=False,
            error_code=ErrorCode.E_IO_ERROR,
            error_message=error_msg,
        )

"""
FS-01 create_directory 插件
版本: 1.0.0
描述: 创建目录（如不存在）
依赖: 标准库 pathlib, logging
"""

import logging
from pathlib import Path

logger = logging.getLogger("libsmall.create_directory")

_MAX_PATH_LENGTH = 1024


def create_directory(dir_path: str) -> tuple[bool, bool]:
    """
    创建目录（如不存在）。

    参数:
        dir_path: 要创建的目录路径

    返回:
        (created, exists)
        - created: 本次是否创建了目录
        - exists:  目录是否存在（创建后）
    """

    # 输入校验
    if not dir_path or not dir_path.strip():
        logger.error("创建目录失败, 路径不能为空")
        return False, False

    dir_path = dir_path.strip()

    if len(dir_path) > _MAX_PATH_LENGTH:
        logger.error(f"创建目录失败 {dir_path}, 路径长度超过 {_MAX_PATH_LENGTH} 字符")
        return False, False

    # 转换为 Path 对象
    path = Path(dir_path)

    try:
        # 检查路径是否已存在
        if path.exists() and path.is_dir():
            logger.debug(f"目录已存在 {dir_path}")
            return False, True

        # 不存在，执行创建
        path.mkdir(parents=True, exist_ok=True)
        logger.debug(f"目录创建成功 {dir_path}")
        return True, True

    except PermissionError:
        logger.error(f"创建目录失败 {dir_path}, 权限不足")
        return False, False

    except OSError as e:
        logger.error(f"创建目录失败 {dir_path}, {e}")
        return False, False

# ⚠️ 插件开发警告清单

在开发任何插件之前，必须逐条对照本文件。违反即返工。

---

## 一、输出必须严格对齐规范

### 🔴 禁止擅自增加字段

规范定义了什么输出，就返回什么输出。**不多不少。**

规范定义

outputs:

  • name: created
    type: bool
  • name: exists
    type: bool

✅ 正确

return False, True

❌ 错误：擅自加了 error_code、error_message

return CreateDirectoryResult(created=False, exists=True, error_code="E_PERMISSION", ...)

TEXT

**判断标准：** 返回值中出现任何规范 `outputs` 里没有的字段，即为违规。

---

## 二、不要过度封装

### 🟡 不使用规范未要求的包装结构

规范没要求 dataclass / NamedTuple / 自定义类，就不要用。
元组能解决的事情，不要造一个类。

✅ 规范说返回 (bool, bool)，就返回元组

return True, True

❌ 没人让你写 dataclass

@dataclass
class CreateDirectoryResult:
...

TEXT

---

## 三、控制代码量

### 🟡 实现体量应与规范复杂度匹配

一个简单函数的插件，核心逻辑不应超过 **50 行**。
如果写到 150 行，一定是加了规范没要求的东西。

**自查方法：** 写完后把代码和规范并排放，逐行对照，删掉所有在规范中找不到依据的代码。

---

## 四、错误处理严格按 internal_logic

### 🔴 异常分支的返回值以规范为准

规范写了 `PermissionError → 返回 (False, False)`，就只返回 `(False, False)` 并记日志。
不要自作主张把错误码塞进返回值。

规范: PermissionError → 记录日志，返回 (False, False)

✅ 正确

except PermissionError:
logger.error(f"创建目录失败 {dir_path}, 权限不足")
return False, False

❌ 错误：返回值里夹带私货

except PermissionError:
return CreateDirectoryResult(
created=False, exists=False,
error_code=ErrorCode.E_PERMISSION, # 规范没让你返回这个
error_message="..." # 规范没让你返回这个
)

TEXT

---

## 五、日志模板必须与规范一致

### 🔴 日志内容逐字对照

规范定义了 `template`，代码中的日志字符串必须包含该模板内容。
测试用 `contains` 匹配，模板关键词不能改。

规范 template: "目录已存在 {dir_path}"

✅ 正确

logger.debug(f"目录已存在 {dir_path}")

❌ 错误：改了措辞

logger.debug(f"路径 {dir_path} 已经存在，跳过创建")

TEXT

---

## 六、日志级别必须与规范一致

### 🔴 DEBUG 就是 DEBUG，ERROR 就是 ERROR

规范: trigger=目录已存在, level=DEBUG

✅ 正确

logger.debug(...)

❌ 错误：升级成了 INFO

logger.info(...)

TEXT

---

## 七、dependencies 为空就只用标准库

### 🔴 不引入第三方依赖

`dependencies: []` 意味着只能用 Python 标准库。
不要引入 `click`、`pydantic`、`attrs` 等任何第三方包。

---

## 八、开发流程检查单

每次提交前逐项打勾：

- [ ] 返回值字段与规范 `outputs` **完全一致**，无增无减
- [ ] 返回值类型与规范一致（bool 就是 bool，不是 Optional[bool]）
- [ ] 异常分支返回值与 `internal_logic` 一致
- [ ] 日志模板关键词与规范 `logs.template` 一致
- [ ] 日志级别与规范 `logs.level` 一致
- [ ] 没有使用规范未要求的类/结构体包装
- [ ] 没有引入规范 `dependencies` 之外的依赖
- [ ] 核心逻辑代码量与规范复杂度匹配，无冗余

---

## 一句话总结

> **规范写什么就实现什么，不多不少，不改不加。**
